var reverseWords = function(str) {
    if(/\S/.test(str)){
        let arr = str.trim().split(' ');
        let result = [];
        arr.forEach(word => {
            if(word !== '') result.push(word);
        })
        return result.reverse().join(' ');
    } else {
        return "";
    }
};
